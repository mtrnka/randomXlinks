Heres is my function that reads a PDB from the database when given a valid PDB accession:

parsePDB <- function(pdbid, pdbFileDir=getwd()) {
    if (file.exists(file.path(pdbFileDir, pdbid))) {
        pdbCode <- file.path(pdbFileDir, pdbid)
    } else {
        pdbCode = str_extract(pdbid, "^[[0-9A-Za-z]]{4}")
    }
    if (is.na(pdbCode)) message(str_c(pdbid, " is not a valid pdb id"))
    fileExtention <- str_extract(pdbid, "(?<=\\.)[[^\\.]]+$")
    if (!fileExtention %in% c("cif", "pdb")) fileExtention <- "pdb"
    parsedPDB <- tryCatch(
        switch(fileExtention,
               "cif" = bio3d::read.cif(pdbCode, verbose=F),
               "pdb" = bio3d::read.pdb(pdbCode, verbose=F)
        ),
        error = function(e) {
            message(str_c("Trouble reading pdb id: ", pdbid, "\n", e))
            return(NA)
        }
    )
    if (!file.exists(file.path(pdbFileDir, pdbid)) & ("pdb" %in% class(parsedPDB))) {
        bio3d::write.pdb(parsedPDB, file.path(pdbFileDir, str_c(pdbCode, ".pdb")))
    }
    ca.ind <- bio3d::atom.select(parsedPDB, "calpha")
    parsedPDB <- parsedPDB$atom[ca.ind$atom,]
    return(parsedPDB)
}

Here are some other functions to measure euclidean distances from parsedPDBs and then a function to sample random crosslinks from a PDB and measure distances:

euclideanDistance <- function(residue1, chain1, residue2, chain2, parsedPDB) {
    coord1 <- parsedPDB[parsedPDB$chain==chain1 & parsedPDB$resno==residue1,
                        c("x","y","z")]
    if (nrow(coord1)==0) coord1 <- NA
    coord2 <- parsedPDB[parsedPDB$chain==chain2 & parsedPDB$resno==residue2,
                        c("x","y","z")]
    if (nrow(coord2)==0) coord2 <- NA
    distance <- sqrt(sum((coord1 - coord2)**2))
    return(round(distance,2))
}

getRandomCrosslinks <- function(parsedPDB, numCrosslinks,
                                resType1="LYS", resType2="LYS",
                                chains=NULL) {
   if (!is.null(chains)) {
      parsedPDB <- parsedPDB %>% filter(chain %in% chains)
   }
   pdb1 <- parsedPDB %>% 
      filter(resid %in% resType1) %>%
      sample_n(numCrosslinks, replace=T) %>%
      select(residue1=resno, chain1=chain)
   pdb2 <- parsedPDB %>%
      filter(resid %in% resType2) %>%
      sample_n(numCrosslinks, replace=T) %>%
      select(residue2=resno, chain2=chain)
   pdb <- cbind(pdb1, pdb2)
   randomDists <- pdb %>% pmap_dbl(
      euclideanDistance, parsedPDB
   )
   return(randomDists)
}
